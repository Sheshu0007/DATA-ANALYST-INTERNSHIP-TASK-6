FROM saless;

2. Total Revenue by Month
SELECT SUM(total_revenue) AS total_revenue, MONTH(`date`) AS months
FROM saless
GROUP BY MONTH(`date`);

3. Total Revenue by Year
SELECT SUM(total_revenue) AS total_revenue, YEAR(`date`) AS year
FROM saless
GROUP BY YEAR(`date`);
